\newcommand{\tableEnv}[2]{
  \begin{rSection}{#1}
    \begin{adjustwidth}{0.0in}{0.1in}
      \begin{tabularx}{\linewidth}{@{} >{\bfseries}l @{\hspace{6ex}} X @{}}
        #2
      \end{tabularx}
    \end{adjustwidth}
  \end{rSection}
}
